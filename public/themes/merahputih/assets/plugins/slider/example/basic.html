jQuery SilverTrack

SilverTrack was designed to be extensible through other plugins.

To create a new track is simple as: 

      
<div class="view-port">
  <div class="slider-container">
    <div class="item"></div>
    <div class="item"></div>
    <!-- ... -->
  </div>
</div>
    
      
.view-port {
  overflow: hidden;
  position: relative;
}
.slider-container {
  position: relative;
  margin: 0;
  padding: 0;
}
.item {
  position: absolute;
}
    
      
var track = $(".slider-container").silverTrack();

// install the plugins you want
track.install(new SilverTrack.Plugins.Navigator({
  prev: $("a.prev", parent),
  next: $("a.next", parent)
}));

track.start();
